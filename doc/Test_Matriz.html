def test_simple
        @m1 = MathExpansion::Matriz_Densa.new(2,2)
        @m2 = MathExpansion::Matriz_Densa.new(2,2)
        @m3 = MathExpansion::Matriz_Densa.new(2,2)

        @m1.set(0,0,1)
        @m1.set(0,1,2)
        @m1.set(1,0,3)
        @m1.set(1,1,4)
        
        @m2.set(0,0,5)
        @m2.set(0,1,6)
        @m2.set(1,0,7)
        @m2.set(1,1,8)

        @m3.set(0,0,6)
        @m3.set(0,1,8)
        @m3.set(1,0,10)
        @m3.set(1,1,12)


        assert_equal(@m3.to_s,(@m1+@m2).to_s)
end
